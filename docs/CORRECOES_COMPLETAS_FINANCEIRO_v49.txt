SOLUÇÃO — CRIAR FUNÇÃO CENTRAL resolveFinancialCycle

  /**
   * resolveFinancialCycle — única fonte de verdade para calcular o ciclo de uma unidade.
   * Todas as procedures financeiras devem usar esta função.
   * Retorna o mesmo intervalo que calcCycleDates, mas centralizado com lookup automático.
   */
  async function resolveFinancialCycle(
    db: Awaited<ReturnType<typeof getDb>>,
    unitId: number,
    refDate: Date
  ): Promise<{ startDate: Date; endDate: Date; label: string; cycle_configured: boolean }> {
    if (!db) {
      const fallback = calcCycleDates(1, 31, refDate);
      return { ...{ startDate: fallback.cycleStart, endDate: fallback.cycleEnd, label: fallback.label }, cycle_configured: false };
    }
    const unitRow = await db
      .select({
        s: units.billing_cycle_start_day,
        e: units.billing_cycle_end_day,
      })
      .from(units)
      .where(eq(units.id, unitId))
      .limit(1);

    const configured = !!(unitRow[0]?.s && unitRow[0]?.e);
    const { cycleStart, cycleEnd, label } = calcCycleDates(
      unitRow[0]?.s,
      unitRow[0]?.e,
      refDate,
    );
    return {
      startDate: cycleStart,
      endDate: cycleEnd,
      label,
      cycle_configured: configured,
    };
  }
