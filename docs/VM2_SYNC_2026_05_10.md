## PASSO 1 — Renomear colunas `visit_key` → `report_key` e `visits_count` → `reports_count`

> **Execute na VM2:** `mysql -u root -p137946 pacs_portal`

```sql
-- 1a. billing_visit_events: visit_key → report_key
-- (a constraint UNIQUE uq_visit_event precisa ser recriada com o novo nome)
ALTER TABLE `billing_visit_events`
  DROP INDEX `uq_visit_event`,
  CHANGE COLUMN `visit_key` `report_key` VARCHAR(300) NOT NULL,
  ADD CONSTRAINT `uq_report_event` UNIQUE (`report_key`);

-- 1b. billing_cycles: total_visits → total_reports
ALTER TABLE `billing_cycles`
  CHANGE COLUMN `total_visits` `total_reports` INT NOT NULL DEFAULT 0;

-- 1c. billing_cycle_doctor_summary: visits_count → reports_count
ALTER TABLE `billing_cycle_doctor_summary`
  CHANGE COLUMN `visits_count` `reports_count` INT NOT NULL DEFAULT 0;

-- 1d. billing_cycle_system_summary: visits_count → reports_count
ALTER TABLE `billing_cycle_system_summary`
  CHANGE COLUMN `visits_count` `reports_count` INT NOT NULL DEFAULT 0;
```

## PASSO 7 — Limpar templates e frases genéricos (seeds)

> **Execute na VM2:** `mysql -u root -p137946 pacs_portal`

```sql
-- Migration 0039: remove templates globais sem dono e frases genéricas
-- Laudos existentes NÃO são afetados

-- Desvincular laudos que referenciam templates genéricos
UPDATE `reports`
SET `template_id` = NULL
WHERE `template_id` IN (
  SELECT `id` FROM `templates`
  WHERE `isGlobal` = TRUE
    AND `owner_user_id` IS NULL
    AND `unit_id` IS NULL
);

-- Deletar templates globais genéricos
DELETE FROM `templates`
WHERE `isGlobal` = TRUE
  AND `owner_user_id` IS NULL
  AND `unit_id` IS NULL;

-- Deletar frases dos grupos genéricos
DELETE FROM `phrases`
WHERE `group_id` IN (1000, 1001, 1002, 1003, 1004);

-- Deletar grupos de frases genéricos
DELETE FROM `phrase_groups`
WHERE `id` IN (1000, 1001, 1002, 1003, 1004);
```
